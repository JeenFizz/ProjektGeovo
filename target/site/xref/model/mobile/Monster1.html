23       * @param heroPos, tileMap
24       * gets the position of Lorann and directs the monster to him.
25       */
26      public MobileOrder getDirection(Point heroPos, IElement[][] tileMap) {
27      	Point Monsterpos = this.getPos().getLocation();
28      	
29  
30          if(Monsterpos.x == heroPos.x) {
31              if(Monsterpos.y > heroPos.y) {
32                  direction = MobileOrder.Left;
33              } else if (Monsterpos.y < heroPos.y) {
34                  direction = MobileOrder.Right;
35              }
36          } else if(Monsterpos.y == heroPos.y) {
37              if(Monsterpos.x > heroPos.x) {
38                  direction = MobileOrder.Up;
39              } else if (Monsterpos.x < heroPos.x) {
40                  direction = MobileOrder.Down;
41              }
42          } else if (Monsterpos.x < heroPos.x) {
43              if(Monsterpos.y > heroPos.y) {
44                  direction = MobileOrder.LeftDown;
45              } else if (Monsterpos.y < heroPos.y) {
46                  direction = MobileOrder.RightDown;
47              }
48          } else if (Monsterpos.y < heroPos.y) {
49              if(Monsterpos.x > heroPos.x) {
50                  direction = MobileOrder.RightUp;
51              } else if (Monsterpos.x < heroPos.x) {
52                  direction = MobileOrder.LeftUp; 
53              }
54          } else if (Monsterpos.y != heroPos.y){
55          	direction = MobileOrder.random();
56          }else if (Monsterpos.x != heroPos.x){
57          	direction = MobileOrder.random();
58          }
59  
60         
61  
62          return direction;
63      }
