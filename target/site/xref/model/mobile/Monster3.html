19      public MobileOrder getDirection(Point heroPos, IElement[][] tileMap) {
20      	
21      	Point Monsterpos = this.getPos().getLocation();
22      	
23      	if(Monsterpos.x == heroPos.x) {
24              if(Monsterpos.y > heroPos.y) {
25                  direction = MobileOrder.Left;
26              } else if (Monsterpos.y < heroPos.y) {
27                  direction = MobileOrder.Right;
28              }
29          } else if(Monsterpos.y == heroPos.y) {
30              if(Monsterpos.x > heroPos.x) {
31                  direction = MobileOrder.Up;
32              } else if (Monsterpos.x < heroPos.x) {
33                  direction = MobileOrder.Down;
34              }
35          } else if (Monsterpos.x != heroPos.x) {
36              if(Monsterpos.y > heroPos.y) {
37                  direction = MobileOrder.Left;
38              } else if (Monsterpos.y < heroPos.y) {
39                  direction = MobileOrder.Right;
40              }
41          } else if (Monsterpos.y != heroPos.y) {
42              if(Monsterpos.x > heroPos.x) {
43                  direction = MobileOrder.Right;
44              } else if (Monsterpos.x < heroPos.x) {
45                  direction = MobileOrder.Left; 
46              }
47  		
48      }
49      	return direction;
50      }
